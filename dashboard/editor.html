  // Preview using about:blank + document.write (robust across browsers)
  function runPreview(){
    try{
      clearLog();
      const html = buildDocument();
      const frame = document.getElementById('preview');

      // Reset src/srcdoc to force a clean document
      frame.removeAttribute('src');
      frame.removeAttribute('srcdoc');

      frame.onload = () => {
        try{
          const win = frame.contentWindow;
          const doc = win.document;
          doc.open(); doc.write(html); doc.close();

          // Console proxy inside the iframe
          try{
            ['log','error','warn'].forEach(fn=>{
              const orig = win.console[fn];
              win.console[fn] = function(...args){
                try { orig.apply(this,args); } catch(_) {}
                log(`[${fn}] ` + args.map(a => (typeof a === 'string' ? a : JSON.stringify(a))).join(' '));
              };
            });
            win.addEventListener('error', (e)=> log(`[error] ${e.message || e.error}`));
            win.console.log('preview loaded');
          }catch(_){}
        }catch(err){
          log(`[error] ${err && err.message ? err.message : err}`);
        }
      };

      // Start with same-origin blank page
      frame.src = 'about:blank';
    }catch(err){
      log(`[error] ${err && err.message ? err.message : err}`);
    }
  }

  // New Tab (inline or blob fallback)
  function openInNewTab(){
    const warn = document.getElementById('popupWarn');
    warn.classList.remove('show');
    try{
      const doc = buildDocument();
      const w = window.open('', '_blank');
      if (w && w.document){
        w.document.open(); w.document.write(doc); w.document.close(); return;
      }
      const blob = new Blob([doc], {type:'text/html'});
      const url = URL.createObjectURL(blob);
      const a = document.createElement('a');
      a.href = url; a.target = '_blank'; a.rel = 'noopener';
      document.body.appendChild(a); a.click(); a.remove();
      setTimeout(()=>URL.revokeObjectURL(url), 30000);
    }catch(e){
      warn.classList.add('show');
      log(`[error] ${e && e.message ? e.message : e}`);
    }
  }

  // Download
  function downloadHTML(){
    const doc = buildDocument();
    const blob = new Blob([doc], {type:'text/html'});
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url; a.download = 'rgztec-snippet.html';
    document.body.appendChild(a); a.click(); a.remove();
    setTimeout(()=>URL.revokeObjectURL(url), 30000);
  }

  function maybeAutoRun(){
    if (document.getElementById('autoRun').checked) runPreview();
  }

  // Shortcut
  window.addEventListener('keydown', (e)=>{
    if ((e.ctrlKey || e.metaKey) && e.key === 'Enter'){ runPreview(); }
  });
